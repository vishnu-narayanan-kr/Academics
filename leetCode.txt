322. CoinChange

class Solution {
    public int coinChange(int[] coins, int amount) {
        int[] numOfCoinsArr = new int[amount + 1];

        for(int subAmount = 1; subAmount < amount + 1; subAmount++) {
            int numOfCoins = -1;

            for(int coin : coins) {
                if(subAmount - coin >= 0) {
                    if(numOfCoinsArr[subAmount - coin] != -1) {
                        if(numOfCoins == -1) {
                            numOfCoins = numOfCoinsArr[subAmount - coin] + 1;
                        } else {
                            numOfCoins = Math.min(numOfCoins, numOfCoinsArr[subAmount - coin] + 1);
                        }
                    }
                }
            }

            numOfCoinsArr[subAmount] = numOfCoins;
        }

        return numOfCoinsArr[amount];
    }
}
